#include <stdio.h>
#include <string.h>

int main(){ 
    char buffer[100];
    
    int i, dot=0, case1=0, case2=0;
    
    printf("Type anything:");
    
    gets(buffer);
    
    for(i = 0; i < strlen(buffer); i++){
        
        if(buffer[i]=='.'){
            dot ++;
        }
        
        if(buffer[i]>= 58 && buffer[i]< 127){
            
            case1++;
        }
        
        if(buffer[i]>=0 && buffer[i]<46 && buffer[i!=47]){
            
            case2++;
        }
    }
    if(dot>1 ||case1>0 ||case2>0){
        
        printf("\nIs not a number!!!\n\n\n\n");
        
    } else{
        
        printf("\nIs a number!!!\n\n\n\n");
    }
    return 0;
}
